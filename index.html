Data Generator Code

This project is designed to generate synthetic datasets for testing purposes. It simulates real-world data, which allows me to experiment with various data processing techniques before applying them to live data. This mock data has been particularly useful for training models and visualizing patterns in larger datasets.

import random
import pandas as pd

# Simulate a diverse dataset with additional attributes
names = ["Stephen", "Mark", "Ben", "Abby", "Emma", "Olivia", "Liam", "Noah", "Sophia", "Mason"]
cities = ["New York", "Los Angeles", "Chicago", "Houston", "Miami", "Dallas", "Austin", "San Francisco"]
departments = ["Engineering", "Marketing", "Sales", "Finance", "HR", "Product Development"]

data = []
for _ in range(800):
entry = {
    "Name": random.choice(names),
    "Age": random.randint(18, 60),
    "Score": random.randint(50, 100),
    "City": random.choice(cities),
    "Department": random.choice(departments),
    "Experience": random.randint(1, 30),
    "Salary": random.randint(40000, 120000),
    "Gender": random.choice(["Male", "Female", "Non-binary"]),
    "Is_Manager": random.choice([True, False]),
    "Date_of_Joining": f"202{random.randint(0,9)}-{random.randint(1,12):02d}-{random.randint(1,28):02d}"
}
data.append(entry)

df = pd.DataFrame(data)
df.to_csv("mock_data_extended.csv", index=False)
print("Mock data saved to mock_data_extended.csv")

Data Visualization Code

This project demonstrates how data visualization can provide insights into the underlying structure of data. By leveraging libraries like Matplotlib and NumPy, I created interactive charts to help better understand data distributions, correlations, and trends over time.

import random
import matplotlib.pyplot as plt
import numpy as np

# Simulate data for visualizations
names = ["David", "Antonio", "Nolan", "Ashton", "Bethany", "Lily", "Ethan", "Isabella"]
scores = [random.randint(50, 100) for _ in names]
age = [random.randint(18, 60) for _ in names]
department = ["Engineering", "Marketing", "Sales", "Finance", "HR", "Product Development"]

# Bar chart for scores
plt.figure(figsize=(14, 7))
plt.bar(names, scores, color=np.random.rand(len(names), 3))
plt.xlabel("Names")
plt.ylabel("Scores")
plt.title("Randomly Generated Scores")
plt.xticks(rotation=90)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()

# Scatter plot for age vs score
plt.figure(figsize=(10, 6))
plt.scatter(age, scores, alpha=0.6, c=np.random.rand(len(names), 3))
plt.xlabel("Age")
plt.ylabel("Scores")
plt.title("Age vs Scores")
plt.grid(True)
plt.tight_layout()
plt.show()
